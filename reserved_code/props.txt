<button (click)="togglePlayerStatus()">Toggle Player Status</button>

export class AppComponent {
  title = 'frontend';
  playerName = 'John Doe';
  playerStats = {
    secondsMs: 1200,
    isActive: false
  };
  togglePlayerStatus() {
    this.playerStats = {
      ...this.playerStats,
      isActive: !this.playerStats.isActive
    };
    console.log('Novo valor de isActive:', this.playerStats.isActive);
  }
}
